חיים נחמן ביאליק
לא הראני אלהים

לא הֶרְאֵנִי אֶלֹהִים בַּחֲלוֹמוֹת לָיְלָה
וְקוֹסֶם לֹאנָחֵשׁ לֵ,

אִיפֹה יִדְבָקֵנִייוֹמִי הָאַחֲרוֹן
וּמַה-דָּמוּת לִקְצִי כִּייאָדֶע:

אִם בְּתוֹךּ אֶהֲלִ עַלייְצועִי ֶגועָה
וּמַחֲמדִּי בֶּּם צַליָד,

עַד-אֶחֶד יבאו, יַחֲנוּ עָלַי דוּמָם,
מִשְׁמֶרֶת אַהֲכָה וּקְדַשָׂה סְבִיב מִטָּתָ,
וּמְנו נָשִׁימוֹתֵי הָאַחֲרוֹנוֹת בְּחִיק אֵל חַיי
כִּמְנוֹת סְגַלּוֹת חֶמְדָה -

אוֹ בְזוּיוּמְנאָץ, זְעוּם אֲלֹהִים וְאָדֶם,
שְׂנוּא חָבֶר וּפָלִיט מִשְׁפָחָה

בְּתוֹדּ מִכְלָא נְטוּשׁ, עַל אֲלָמַת תֶּבֶה
אֶתנַפְשִׁי, חֲלָלָה וּטְמְאָה, אֶפַּחו

לֹא יַעֲמֹד אִישׁ עַל וְצִיאַת נִשְׁמְִי
ְיִד לֹאיתְרְעַד עַל-עֵיני הַכָּבוֹת.

אוֹ אוּלַי בָּעֶם רְעָבִי וְּמְאי

לַחַיִּים וּלֶכֶלמִנְעַמּוֹתֶם

בִּשְׁאָט בְּנָפֶשׁ, עַליאַף הַּוֹצֵר וַחֲמָתו,
בְּמַתַּת יָדוֹ אֶבָעָם,

ּכְהַשְלּ נַעַל מְגֹאֶלֶת מַרְגָל

 

ַפְשִׁי לָרַגְלָיו אַשְׁלֵיכָה.

וְאוּלֵי מֵרֹב הָארֵךּ רוּחַ אֶמִקָּה
וּמְרֹב יָחָל וְהַחֲרשׁע

ְנִשְׁפּכָה נִפְשִׁי ארְצָה בִּמְרָרָתָ,
ּבִדְמִי לִבִּ אָפְלְטָנָה.

או שֶׁמָּא כְּמִרְגָּלִית, וְנֹגָהּ עוֹלֶם לָהּ,
שִם דִּמְעָתִי הָאַחֲרוֹנָה תֵּפֹל,

ְרְעֵדָה, וְזָרְחָה אַחֲרֵי דּוֹרוֹת רַבִּים
לָעֵינַיִם לאירָאוּנָי.

וְאוּלֵי כַּפּרְפָרה, מִסָּבִיב לַשַׁלְהֶבֶת
רקד וּפַזז תֵּצֵא נַפָשֵי;

או כְשַלְהֶבֶת הַנֵּר עַצְמָה, לִפְני כְלוֹת שַׁמְנָהּ
ְּחַבְלֵי מוֹתָה תְּפַרְפָּר,

וְהִבְהֲכָה וְעֶשְׁנָה עוֹד יָמִים רַבִּים,
תִּעְּוּעִים לָעָיָה

עֲדִי ָפְלָהּ פִּתֶאם אֶלִיתְּהוֹם אֲפלָה

 

בְכֶל-לַחֲבוֹתֶיהָ תִּתְַקֵּ

ְהַפִיצָה בָּעָבִים לַפִּידִי אֶשָׂהּ
וּמְדוּרוֹתֶיהָ בְּרֶאשִׁי הָרִים,
וּפְקוּחוֹת לִרְוְחָה אַלְפִי עִינִים
לִקְָרֵאֹת זְהָרָהּ הָאַחֲרוֹן יִשְׁתָּאוּ
וּמִי יוֹדֵע, אוּלַי יִתְאַכְוֶר לי אֲלִי -
ּבְעוֹדְנִּי חי אָמוּתָה,

וְֶדְרוּ אֶתנַפְשִׁ בְּתְַרִיכי יה

גבוה מן הפחד אשיר
